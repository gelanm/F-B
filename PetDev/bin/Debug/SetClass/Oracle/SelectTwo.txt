/// <summary>
/// 根据条件查询数据数据
/// </summary>
/// <param name="p_hasParameter">
/// 更新条件为:
/// 1: POSID
/// 2: AID
/// </param>
/// <returns>返回查询数据</returns>
    public  DataTable   GetInfo (Hashtable p_hasParameter) 
    {
	string strSQL ="";
	strSQL ="select  S_POSITION.POSID ,S_POSITION.CTMID ,S_POSITION.CJOBNAME ,S_POSITION.EJOBNAME ,S_POSITION.INDUSTRYTYPE1 ,S_POSITION.INDUSTRYTYPE2 ,S_POSITION.FUNCTYPE1 ,S_POSITION.JOBAREA ,S_POSITION.TERM ,S_POSITION.PROVIDESALARY ,S_POSITION.SEX ,S_POSITION.AGEFROM ,S_POSITION.AGETO ,S_POSITION.WORKYEAR ,S_POSITION.DEGREEFROM ,S_POSITION.FL1 ,S_POSITION.FLEVEL1 ,S_POSITION.FL2 ,S_POSITION.FLEVEL2 ,S_POSITION.MAJOR1 ,S_POSITION.MAJOR2 ,S_POSITION.ITSKILL1 ,S_POSITION.ITLOWER1 ,S_POSITION.ITUPPER1 ,S_POSITION.ITSKILL2 ,S_POSITION.ITLOWER2 ,S_POSITION.ITUPPER2 ,S_POSITION.JOBEMAIL ,S_POSITION.CJOBINFO ,S_POSITION.EJOBINFO ,S_POSITION.JOBSRC ,S_POSITION.OPERATOR ,S_POSITION.BRANCH ,S_POSITION.POSTDATE ,S_POSITION.STATUS ,S_POSITION.POSCODE ,S_POSITION.FUNCTYPE2 ,S_POSITION.ISSHARE ,S_POSITION.JOBNUM ,S_POSITION.CVLANGUAGE ,S_POSITION.ISSENDEMAIL ,S_POSITION.FLTID ,S_POSITION.REFID1 ,S_POSITION.REFID2 ,S_POSITION.FAILRECEIVE ,S_POSITION.HRUID ,S_POSITION.POSNAME ,S_POSITION.UPDATER ,S_POSITION.UPDATEDATE ,S_REMARK_A.AID ,S_REMARK_A.QID ,S_REMARK_A.HRUID ,S_REMARK_A.USERID ,S_REMARK_A.ANSWER ,S_REMARK_A.PDATE ,S_REMARK_A.ITFLOWID ,S_REMARK_A.ITID ,S_REMARK_A.CTMID from S_POSITION where   S_POSITION.POSID=S_REMARK_A.AID and S_POSITION.POSID=:POSID and S_REMARK_A.AID=:AID";
OracleParameter[] objParameter = new OracleParameter[2];
 
 objParameter[0] = new OracleParameter(":POSID", System.Data.OracleClient.OracleType.Number , 10);
 objParameter[0].Value = p_hasParameter["POSID"];
 
 objParameter[1] = new OracleParameter(":AID", System.Data.OracleClient.OracleType.Number , 10);
 objParameter[1].Value = p_hasParameter["AID"];
 
	return this.ExecuteReadTable(strSQL, objParameter);
   }
